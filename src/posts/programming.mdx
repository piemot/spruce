```py key=value
I'm probably going to talk a lot about code, so
hopefully this ends up looking nice!

# Create a list
L = []
L = list()
L = ['a', 'b', 'c', 'd', 'z']

L = ['a']

# Add items to end
L.append('aa') # ['a', 'aa']

# Insert items at index

# The new item is inserted __before__ the index -
# you can remember this because the index of the new item
# is the same as the index you pass in.
L.insert(0, 'inserted') # ['inserted', 'a', 'aa']
L.insert(2, 'new') # ['inserted', 'a', 'new', 'aa']

# Clear list
L.clear()

# Remove an item by value
L = ['a', 'b', 'c']
L.remove('b') # ['a', 'c']
# This only removes __the first__ item.
L = ['a', 'b', 'a']
L.remove('a') # ['b', 'a'] -- the second 'a' still exists

# It also raises an error if it doesn't exist.
L = ['a', 'b']
```
